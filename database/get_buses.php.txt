<?php
include 'db.php';

$source = $_GET['source'] ?? '';
$destination = $_GET['destination'] ?? '';

$sql = "SELECT * FROM Buses WHERE Source LIKE '%$source%' AND Destination LIKE '%$destination%'";
$result = $conn->query($sql);

$buses = [];

    while ($row = $result->fetch_assoc()) {

    }
}
echo json_encode($buses);
